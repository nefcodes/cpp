#include "car.h"
#include<iostream>


car::car()
{
	fuel = 0;
	speed = 0;
	passengers = 0;
}

car::~car()
{
	std::cout << "Destructor is invoked.";
}

void car::fillfuel(float amount)
{
	fuel = amount;
}

void car::accelerate()
{
	speed++;
	fuel -= 0.5f;
}

void car::brake()
{
	speed = 0;
}

void car::addpassengers(int count)
{
	passengers = count;
}

void car::dashboard()
{
	std::cout << "Fuel: " << fuel << std::endl;
	std::cout << "Speed: " << speed << std::endl;
	std::cout << "passengers: " << passengers << std::endl;
}
